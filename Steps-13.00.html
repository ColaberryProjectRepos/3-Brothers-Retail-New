Creating Measures

Create a measures table and fill it with the following measures.

 

1.  Arrow ItemsSold MOM% =

SWITCH(

    TRUE(),

    ISBLANK([Items Sold MoM%]) || [Items Sold MoM%] = 0, "",

    [Items Sold MoM%] < 0, UNICHAR(9660),

    UNICHAR(9650))

 

2. Arrow Profit Margin MOM% =

SWITCH(

    TRUE(),

    ISBLANK([Profit Margin MOM%]) || [Profit Margin MOM%] = 0, "",

    [Profit Margin MOM%] < 0, UNICHAR(9660),

    UNICHAR(9650))

 

3.  Arrow Profit Margin YoY% =

SWITCH(

    TRUE(),

    ISBLANK([Profit Margin YoY%]) || [Profit Margin YoY%] = 0, "",

    [Profit Margin YoY%] < 0, UNICHAR(9660),

    UNICHAR(9650))

 

4. Arrow Sales MOM% =

SWITCH(

    TRUE(),

    ISBLANK([Sales MoM%]) || [Sales MoM%] = 0, "",

    [Sales MoM%] < 0, UNICHAR(9660),

    UNICHAR(9650))

 

5.  IBC ItemsSold MoM% =

IF(

    ISBLANK([Items Sold MoM%]) || [Items Sold MoM%] = 0,

    " ",

    FORMAT([Items Sold MoM%], "0%"))

 

6. IBC Profit Margin =

IF(

    ISBLANK([Profit Margin]) || [Profit Margin] = 0,

    " ",

    FORMAT([Profit Margin], "0%"))

 

7. IBC Profit Margin YoY% =

IF(

    ISBLANK([Profit Margin YoY%]) || [Profit Margin YoY%] = 0,

    " ",

    FORMAT([Profit Margin YoY%], "0%"))

 

8. IBC Sales MoM% =

IF(

    ISBLANK([Sales MoM%]) || [Sales MoM%] = 0,

    " ",

    FORMAT([Sales MoM%], "0%"))

 

9. Items Sold MoM% = DIVIDE(([Total Items Sold] -[TotalItemsSold LM]),[TotalItemsSold LM])

 

10. Profit = SUM(SuperStore_Sales_Dataset[Profit])

 

11. Profit LY = CALCULATE(SUM(SuperStore_Sales_Dataset[Profit]), DATEADD('Calendar'[Date],-1,YEAR))

 

12. Profit Margin = DIVIDE([Profit],[Total Sales],0)

 

13. Profit Margin LM = CALCULATE([Profit Margin],DATEADD('Calendar'[Date],-1,MONTH))

 

14. Profit Margin LY = CALCULATE([Profit Margin],DATEADD('Calendar'[Date],-1,YEAR))

 

15. Profit Margin MOM% = DIVIDE(([Profit Margin]- [Profit Margin LM]),[Profit Margin LM])

 

16. Profit Margin YoY% = DIVIDE(([Profit Margin]- [Profit Margin LY]),[Profit Margin LY])

 

17.  Sales MoM% = DIVIDE(([Total Sales]-[TotalSales LM]),[TotalSales LM])

 

18. Total Items Sold = SUM(SuperStore_Sales_Dataset[Quantity])

 

19. Total Sales = SUM(SuperStore_Sales_Dataset[Sales])

 

20. TotalItemsSold LM = CALCULATE(SUM(SuperStore_Sales_Dataset[Quantity]),  DATEADD('Calendar'[Date],-1,MONTH))

 

21. TotalSales LM = CALCULATE(SUM(SuperStore_Sales_Dataset[Sales]),  DATEADD('Calendar'[Date],-1,MONTH))
